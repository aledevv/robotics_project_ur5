81def store_blocks(data):
82
83 '''
84 @brief takes blocks data from json object from YOLO detection to create a list of blocks
85 @param JSON/dictionary object
86 @retval a list of Blocks
87 '''

109def detection(img_path):
110 '''
111 @brief It crops the ZED-cam image into the Region of Interest (only the table), block detection with YOLO follows
