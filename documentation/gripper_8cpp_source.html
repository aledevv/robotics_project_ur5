279MatrixX3d get_trajectory(Vector3d final_position){
280
281 int stationary_points_num = 7;
282 Matrix<double, 7, 3> stationary_points;
283 stationary_points << 0.3, 0.1, 0.5,
284 0.4, 0, 0.5,
285 0.3, -0.3, 0.5,
286 0, -0.4, 0.5,
287 -0.3, -0.3, 0.5,
288 -0.4, 0, 0.5,
289 -0.3, 0.1, 0.5;
290
291
292
293 V6d joint_state = get_joint_state(get_robot_values());
294 M4d transformation_matrix = mwtb() * direct_kin(joint_state) * gripper_frame();
295 V3d init_position = transformation_matrix.block(0, 3, 3, 1);
296
297
298 int starting_position = -1;
299 double min_distance = -1;
300 double possible_min_distance;
301 for (int i = 0; i < stationary_points_num; i++)
302 {
303 possible_min_distance = abs(stationary_points(i, 0) - init_position(0));
304 possible_min_distance = possible_min_distance + abs(stationary_points(i, 1) - init_position(1));
305 if (min_distance == -1 || possible_min_distance < min_distance)
306 {
307 min_distance = possible_min_distance;
308 starting_position = i;
309 }
310 }
311
312
313 int ending_position = -1;
314 min_distance = -1;
315 for (int i = 0; i < stationary_points_num; i++)
316 {
317 possible_min_distance = abs(stationary_points(i, 0) - final_position(0));
318 possible_min_distance = possible_min_distance + abs(stationary_points(i, 1) - final_position(1));
319 if (min_distance == -1 || possible_min_distance < min_distance)
320 {
321 min_distance = possible_min_distance;
322 ending_position = i;
323 }
324 }
325
326
327 bool trajectory_built = false;
328 int i = starting_position;
329
330 MatrixX3d trajectory;
331
332 do
333 {
334 trajectory.conservativeResize(trajectory.rows() + 1, trajectory.cols());
335 trajectory.row(trajectory.rows() - 1) = V3d {stationary_points(i, 0), stationary_points(i, 1), stationary_points(i, 2)};
336 if (i == ending_position) trajectory_built = true;
337 if (ending_position < starting_position) i = i - 1; else i = i + 1;
338 }
339 while (!trajectory_built);
340
341 trajectory.conservativeResize(trajectory.rows() + 1, trajectory.cols());
342 trajectory.row(trajectory.rows() - 1) = V3d {final_position(0), final_position(1), final_position(2)};
343
344 return trajectory;
