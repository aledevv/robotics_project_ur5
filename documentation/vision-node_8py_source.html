184def find_pose(point_cloud, block):
185'''
186 @brief Function to compute object pose. Basically, the block is "sliced" at a specific height to find the 2
187 useful points:
188 - x_min: the nearest point the camera
189 - y_min: the rightmost point (near the arm)
190 - y_max: the leftmost point (furthest from the arm)
191 these points are used to calculate yaw angle
192 @param point cloud and the block
193 @retval yaw angle (radiant)
194'''
195
196 # CONSTANTS (used only here)
197 table_height = 0.88
198 target_height = 0.012 + table_height
199
200 selected_points = [] # list of points at the target_height value
201
202 print("Finding pose...")
203
204 min_x = Point()
205 min_y = Point()
206 max_y = Point()
207
208 # scan point cloud of the bounding box
209 for x in range(int(block.x1), int(block.x2)):
210 for y in range(int(block.y1), int(block.y2)):
211 for data in point_cloud2.read_points(point_cloud, field_names=['x', 'y', 'z'], skip_nans=True, uvs=[(x,y)]):
212 point_coords = [data[0], data[1], data[2]]
213
214 # transform current point to world coordinate
215 point_world = R_cloud_to_world.dot(point_coords) + x_camera + base_offset
216
217 # check whether it belongs to target height
218 if abs(point_world[0, 2] - target_height) <= 0.001:
219
220 current_coords = (point_world[0, 0], point_world[0, 1], point_world[0, 2])
221 selected_points.append(current_coords)
222
223 if len(selected_points) == 1: # if it is the first point store it as the min_x, min_y
224 color_pixel(x, y, 'red')
225 min_x.set(point_world[0, 0], point_world[0, 1], point_world[0, 2], (x, y))
226 min_y.set(point_world[0, 0], point_world[0, 1], point_world[0, 2], (x, y))
227 max_y.set(point_world[0, 0], point_world[0, 1], point_world[0, 2], (x, y))
228 else:
229 color_pixel(x, y, 'red')
230 min_x.is_min_x(current_coords, (x, y))
231 min_y.is_min_y(current_coords, (x, y))
232 max_y.is_max_y(current_coords, (x, y))
233
234 # Print 3 vertices info
235 #min_x.info()
236 #min_y.info()
237 #max_y.info()
238
239 # Show in yellow points of interest
240 color_pixel(min_x.px[0], min_x.px[1], 'yellow')
241 color_pixel(min_y.px[0], min_y.px[1], 'yellow')
242 color_pixel(max_y.px[0], max_y.px[1], 'yellow')
243
244 x,y = choose_side(min_x, min_y, max_y)
245
246 # Finding yaw angle
247 yaw = math.atan2(x - min_x.x, y - min_y.y)
248
249 return yaw

252def choose_side(x_min, y_min, y_max):
253 '''
254 @brief function calculate longest side of the block in order to compute the angle along it

257
258 if ((x_min.x - y_min.x)**2 + (x_min.y - y_min.y)**2) > ((x_min.x - y_max.x)**2 + (x_min.y - y_max.y)**2):
259 return y_min.x, y_min.y
